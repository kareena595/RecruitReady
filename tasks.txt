import os
import json
from dotenv import load_dotenv
from pydantic import BaseModel, Field
from google.adk.agents import LlmAgent
from google.adk.runners import Runner
from google.adk.sessions import InMemorySessionService
from google.genai import types
from google import genai

# --- Load API key ---
load_dotenv()
api_key = os.getenv("GOOGLE_API_KEY")
if not api_key:
    raise ValueError("Missing GOOGLE_API_KEY environment variable.")
os.environ["GOOGLE_API_KEY"] = api_key

# Quick connection test (optional)
client = genai.Client(api_key=api_key)
print("âœ… Connected to Gemini:", client.models.generate_content(
    model="gemini-2.0-flash", contents="Connection test successful!"
).text)

# --- Constants ---
APP_NAME = "agent_comparison_app"
USER_ID = "test_user_456"
SESSION_ID_TOOL_AGENT = "session_tool_agent_xyz"
SESSION_ID_SCHEMA_AGENT = "session_schema_agent_xyz"
MODEL_NAME = "gemini-2.5-flash"

# --- Schemas ---
class CountryInput(BaseModel):
    country: str = Field(description="The country to get information about.")

class CapitalInfoOutput(BaseModel):
    capital: str = Field(description="The capital city of the country.")
    population_estimate: str = Field(description="Estimated population of the capital city.")

# --- Tool ---
def get_capital_city(country: str) -> str:
    """Retrieve the capital city of a given country."""
    print(f"-- Tool Call: get_capital_city('{country}') --")
    capitals = {
        "united states": "Washington, D.C.",
        "canada": "Ottawa",
        "france": "Paris",
        "japan": "Tokyo",
    }
    return capitals.get(country.lower(), f"Sorry, I couldn't find the capital for {country}.")

# --- Agents ---
capital_agent_with_tool = LlmAgent(
    model=MODEL_NAME,
    name="capital_agent_tool",
    description="Retrieves the capital city using a specific tool.",
    instruction=(
        "You are a helpful agent that provides the capital city of a country using a tool.\n"
        "The user will provide the country name as JSON like {'country': 'country_name'}.\n"
        "1. Extract the country.\n"
        "2. Use get_capital_city tool.\n"
        "3. Respond clearly to the user."
    ),
    tools=[get_capital_city],
    input_schema=CountryInput,
    output_key="capital_tool_result",
)

structured_info_agent_schema = LlmAgent(
    model=MODEL_NAME,
    name="structured_info_agent_schema",
    description="Provides capital and population estimate in JSON format.",
    instruction=f"You must return JSON matching: {json.dumps(CapitalInfoOutput.model_json_schema(), indent=2)}",
    input_schema=CountryInput,
    output_schema=CapitalInfoOutput,
    output_key="structured_info_result",
)

# --- Sessions + Runners ---
session_service = InMemorySessionService()
capital_runner = Runner(agent=capital_agent_with_tool, app_name=APP_NAME, session_service=session_service)
structured_runner = Runner(agent=structured_info_agent_schema, app_name=APP_NAME, session_service=session_service)

# --- Export all important objects for tempApp.py ---

    "capital_runner", "structured_runner",
    "capital_agent_with_tool", "structured_info_agent_schema",

# tempApp.py
import asyncio
import json
from test.agent import (
    APP_NAME, USER_ID, SESSION_ID_TOOL_AGENT, SESSION_ID_SCHEMA_AGENT,
    capital_runner, structured_runner,
    capital_agent_with_tool, structured_info_agent_schema,
    session_service, types
)

async def call_agent_and_print(runner_instance, agent_instance, session_id, query_json):
    print(f"\n>>> Calling {agent_instance.name} with query: {query_json}")
    user_content = types.Content(role='user', parts=[types.Part(text=query_json)])
    final_response_content = "No final response received."

    async for event in runner_instance.run_async(
        user_id=USER_ID, session_id=session_id, new_message=user_content
    ):
        if event.is_final_response() and event.content and event.content.parts:
            final_response_content = event.content.parts[0].text

    print(f"<<< {agent_instance.name} Response: {final_response_content}")

    current_session = await session_service.get_session(
        app_name=APP_NAME, user_id=USER_ID, session_id=session_id
    )
    stored_output = current_session.state.get(agent_instance.output_key)

    print(f"--- Session State [{agent_instance.output_key}]: ", end="")
    try:
        print(json.dumps(json.loads(stored_output), indent=2))
    except Exception:
        print(stored_output)
    print("-" * 40)

async def main():
    print("--- Creating Sessions ---")
    await session_service.create_session(app_name=APP_NAME, user_id=USER_ID, session_id=SESSION_ID_TOOL_AGENT)
    await session_service.create_session(app_name=APP_NAME, user_id=USER_ID, session_id=SESSION_ID_SCHEMA_AGENT)

    print("--- Testing Agent with Tool ---")
    await call_agent_and_print(capital_runner, capital_agent_with_tool, SESSION_ID_TOOL_AGENT, '{"country": "France"}')

    print("\n--- Testing Agent with Schema ---")
    await call_agent_and_print(structured_runner, structured_info_agent_schema, SESSION_ID_SCHEMA_AGENT, '{"country": "Japan"}')

if __name__ == "__main__":
    asyncio.run(main())
